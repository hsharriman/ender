// ------------ Triangle Reasons ------------ 
// iscosceles triangle
reason isosceles(con_seg) -> isosceles

// side-angle-side triangle congruence
reason sas(con_seg, con_ang, con_seg) -> con_tri

// side-side-side triangle congruence
reason sss(con_seg, con_seg, con_seg) -> con_tri

// angle-side-angle triangle congruence
reason asa(con_ang, con_seg, con_ang) -> con_tri

// angle-angle-side triangle congruence
reason aas(con_ang, con_ang, con_seg) -> con_tri

// right-hypotenuse-leg triangle congruence
reason rhl(con_right, con_seg, con_seg) -> con_tri

// corresponding parts of congruent triangles are congruent
reason cpctc(con_tri) -> con_seg, con_ang


// ------------ Line Reasons ------------ 
// definition of intersecting lines
reason intersect_seg() -> intersect_seg

// reflexive property: any segment is congruent to itself
reason reflex_s() -> con_seg


// ------------ Angle Reasons ------------ 
// right angle: the angle formed by two perpendicular (perp) lines is 90 degrees
reason right(perp) -> right

// reflexive property: any angle is congruent to itself
reason reflex_a() -> con_ang

// vertical angles
reason vert_ang(intersect_seg) -> con_ang


// ------------ not implemented ------------ 
// converse of alternate interior angles (given that alternate interior angles are congruent, can conclude that lines are parallel)
reason altint_conv(con_ang, transversal) -> para

// alternate interior angles (given that parallel lines crossed by a transversal, can conclude alternate interior angles are congruent)
reason altint(para, transversal) -> con_ang 

// definition of perpendicular lines
reason perp(right, on_line) -> perp

// definition of midpoint
reason midpt(con_seg, on_line) -> midpt

// converse of definition of midpoint
reason midpt_conv(midpt) -> con_seg, on_line

// definition of rectangle
reason rectangle(con_seg) -> rectangle

// equilateral triangle
reason equilateral(con_seg, con_seg) -> equilateral

// definition of parallelogram (rectangle is type of parallelogram)
reason paralellogram1(rectangle) -> parallelogram

// definition of parallelogram (parallelogram has 2 sets of parallel sides)
reason paralellogram2(para) -> parallelogram

// angle bisector
reason ang_bisect(con_ang) -> ang_bisect

// angle-angle-angle triangle similarity
reason aaa(con_ang, con_ang, con_ang) -> sim_tri
