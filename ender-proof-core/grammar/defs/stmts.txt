// ------ Statements that can only appear in the premises section of a proof ----- 
premises
// point p is on the segment s
stmt on_line(Segment s, Point p)

// the segment from s1p1 -> s1p2 is crossed by t1, and the segment from s2p1 -> s2p2
// is crossed by t2. t1 -> t2 is the transversal.
// NOTE: assumes that s1p1 -> s1p2 and s2p1 -> s2p2 are both chosen wrt the same direction (such as from left to right)
//           t1
//  x--------x-----x
// s1p1     /     s1p2
//        /
//  x----x------------x
// s2p1 t2           s2p2
stmt transversal(Point s1p1, Point s1p2, Point t1, Point s2p1, Point s2p2, Point t2) // Point s1p1 -> s1p2 = seg1
end_premises

// ----- Supported statements with unidirectional substitutability ----- 
// Congruence statements - con_right can substitute for con_ang, but not vice versa
group congruent_angs { base: con_ang, extensions: con_right }

// Point-on-line statements - midpt can substitute for on_line, but not vice versa  
group point_on_line { base: on_line, extensions: midpt }

// Individual statements
stmt right(Angle a) 

// segment s1 and s2 intersect at point p
stmt intersect_seg(Segment s1, Segment s2, Point p)

// segment s1 is congruent to s2
stmt con_seg(Segment s1, Segment s2)

// angle a1 is congruent to a2
stmt con_ang(Angle a1, Angle a2)

// triangle t1 is congruent to t2
stmt con_tri(Triangle t1, Triangle t2)

// right angles a1 and a2 are congruent
stmt con_right(Angle a1, Angle a2)

// segment s1 is parallel to s2
stmt para(Segment s1, Segment s2)

// triangle t is isosceles (2 sides are equal length)
stmt isosceles(Triangle t)

// segment s1 and s2 are perpendicular
stmt perp(Segment s1, Segment s2)

// p is the midpoint of s
stmt midpt(Segment s, Point p)

// s bisects angle a
stmt ang_bisect(Angle a, Segment s)

// q is a rectangle
stmt rectangle(Quad q)

// q is a parallelogram
stmt parallelogram(Quad q)

// t is an isosceles triangle
stmt isosceles(Triangle t)

// s1 is perpendicular to s2
stmt perp(Segment s1, Segment s2)
